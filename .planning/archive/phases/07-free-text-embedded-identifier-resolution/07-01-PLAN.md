// Anchored existing regexes — only match BARE identifiers (the gap we fix):
const DOI_ONLY_REGEX =
  /^(?:(?:https?:\/\/)?(?:dx\.)?doi\.org\/|(?:https?:\/\/)?doi:)?10\.\d{4,}\/[^\s]+$/i;
const PMID_REGEX = /^PMID:\s*(\d{1,8})$/i;

// DOI normalization — strips trailing punctuation .,;:) and a leading doi: prefix.
// PRESERVES the https://doi.org/ form (that is stripped later by *ForLookup).
function normalizeDoiInput(value) {
  return value.trim()
    .replace(/[).,;:\s]+$/u, '')
    .replace(/^(?:https?:\/\/)?doi:/iu, '');
}

// PMID backend consumes value via normalizePmidInput which strips /^PMID:\s*/.
