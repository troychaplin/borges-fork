Output: Three hooks each gain a `computeExportStrings(cslArray, citationStyle)` call after `formatBibliographyEntries` completes, with the result merged into each citation object before `setAttributes`.

async function computeExportStrings(cslObjects, citationStyle) {
  // Returns an array of { exportBibtex, exportBiblatex } indexed by position.
  // Uses single-citation wrappers â€” buildBibtexExportContent accepts a citations array
  // but for per-citation storage we pass each CSL as a single-item array:
  const results = await Promise.all(
    cslObjects.map(async (csl) => {
      try {
        const [bibtex, biblatex] = await Promise.all([
          buildBibtexExportContent([{ csl }], citationStyle),
          buildBiblatexExportContent([{ csl }], citationStyle),
        ]);
        return { exportBibtex: bibtex, exportBiblatex: biblatex };
      } catch {
        return { exportBibtex: '', exportBiblatex: '' };
      }
    })
  );
  return results;
}
